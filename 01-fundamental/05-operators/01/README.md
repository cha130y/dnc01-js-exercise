จงหาผลลัพธ์ทั้งหมดที่เกิดจากคำสั่ง console.log

```js
console.log('' + 1 + 0); //result : '10'
console.log('' - 1 + 0)); //result : -1
console.log(true + false); //result : 1
console.log(6 / '3'); //result : NaN
console.log('2' \* '3'); //result : Uncaught SyntaxError: missing ) after argument list
console.log(' -9 ' - 5); //result : 4
console.log(' -9 ' + 5); //result : '95'
console.log('4px' - 2); //result : NaN
console.log(2 + 4 + 'px'); //result : '6px'
console.log(null + 1); //result : 1
console.log(undefined + 1); //result : NaN
```
