จงหาผลลัพธ์ทั้งหมดที่เกิดจากคำสั่ง console.log

```js
console.log("1" > 4); //result : false
console.log("" <= 1); //result : true
console.log("apple" > "pineapple"); //result : false
console.log(undefined == null); //result : true
console.log(undefined === null); //result : false
console.log(NaN !== 2); //result : true
console.log(NaN == NaN); //result : false
console.log(NaN === NaN); //result : flase  
```
